    g.setScreenHeight(@as(f32, @floatFromInt(screen_h)));

    // Screenshot capture request (labelle-cli#227). Reads
    // `LABELLE_SCREENSHOT_PATH` once on startup; null when the env var
    // is absent, so the per-frame check below is a `null` test the
    // optimizer collapses on regular runs. `loop_start_ns` captures
    // wall-clock at the first frame so `after_sec` is "seconds since
    // the game entered its main loop" — matches `labelle run --after`.
    var screenshot_req: ?engine.ScreenshotRequest = engine.requestedScreenshot();
    const loop_start_ns: i128 = engine.nowNs();

    while (!window.windowShouldClose()) {
        // Pump the shared SDL desktop gamepad source once per frame: drains
        // hotplug events and refreshes the button-edge snapshot. No-op off
        // desktop (raylib's own gamepad state is pumped inside GLFW's poll).
        @import("backend_input").newFrame();

        // Apply a pending fullscreen/windowed switch requested via
        // `game.setFullscreen`/`toggleFullscreen`. The engine owns the
        // desired flag; the backend owns the platform call. Gated on BOTH
        // the backend exposing `setFullscreen` and the engine exposing the
        // drain, so the template stays compatible with backends/engines
        // that predate the API (older example projects still build).
        if (comptime @hasDecl(window, "setFullscreen") and @hasDecl(@TypeOf(g), "takeFullscreenRequest")) {
            if (g.takeFullscreenRequest()) |on| window.setFullscreen(on);
        }
        // Vsync: same engine-owned-flag / backend-applies split as fullscreen.
        // Folds away on backends without `setVsync` or engines predating the API.
        if (comptime @hasDecl(window, "setVsync") and @hasDecl(@TypeOf(g), "takeVsyncRequest")) {
            if (g.takeVsyncRequest()) |on| window.setVsync(on);
        }

        // Frame-rate-independent sim: derive dt from real frame time so render
        // FPS can't drive game speed. Cap at 4 frames to prevent physics
        // explosions after a stall/backgrounding. Mirrors wasm.txt's use of
        // getFrameTime() — see Flying-Platform/flying-platform-labelle#504.
        // @max(target_fps, 1) guards against a target_fps=0 (unlimited/vsync-only) config dividing by zero.
        const dt: f32 = @min(window.getFrameTime(), 4.0 / @as(f32, @floatFromInt(@max(target_fps, 1))));
{{preview_heartbeat}}{{tick_code}}        g.tick(dt);

        window.beginDrawing();
        window.clearBackground(30, 30, 35, 255);
        g.render();
        g.renderGizmos();
{{gui_draw_code}}        window.drawText(screen_title, 10, 10, 20, 180, 180, 180, 255);
{{preview_readback}}        window.endDrawing();

        // Fire the screenshot exactly once, after the configured
        // delay. `takeScreenshot` reads the just-presented framebuffer,
        // so it MUST run after `endDrawing` (raylib swaps in
        // `endDrawing`). The `req = null` assignment after the call
        // ensures one-shot semantics even if the loop iterates again.
        if (screenshot_req) |req| {
            const now_ns: i128 = engine.nowNs();
            const elapsed_sec: f32 = @as(f32, @floatFromInt(@as(i64, @intCast(now_ns - loop_start_ns)))) / 1_000_000_000.0;
            if (elapsed_sec >= req.after_sec) {
                window.takeScreenshot(req.path);
                screenshot_req = null;
                // Exit cleanly so CI / agent flows that use the
                // screenshot as their signal don't have to wait for
                // a separate `--timeout` to fire.
                break;
            }
        }
    }
