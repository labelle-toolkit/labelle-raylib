// --- WASM/Emscripten main loop ---
// Raylib's web target requires emscripten_set_main_loop instead of a while loop.
// The game tick + render is driven by a per-frame callback.

var g: AssembledGame = undefined;
{{hooks_init_block}}
var g_allocator: std.mem.Allocator = undefined;
var runner: Runner = undefined;

fn gameFrame() callconv(.c) void {
    const dt: f32 = @floatCast(window.frameDuration());
{{preview_heartbeat}}{{tick_code}}    g.tick(dt);

    window.beginFrame();
    window.clearBackground(245, 245, 245, 255);
    g.render();
    g.renderGizmos();
{{gui_draw_code}}    window.drawText(screen_title, 10, 10, 20, 80, 80, 80, 255);
    window.endFrame();
}

// Hand-rolled extern shim instead of `@cImport(emscripten.h)` — Zig 0.16's
// translate-c rejects recent emsdk headers (multi-arg
// `__attribute__((deprecated("msg1","msg2")))`, translate-c issue #306).
// We only need the main-loop entry; emcc resolves the symbol at link time.
extern "c" fn emscripten_set_main_loop(
    func: *const fn () callconv(.c) void,
    fps: c_int,
    simulate_infinite_loop: c_int,
) void;

pub fn main() !void {
    // Use c_allocator for Emscripten — it delegates to emscripten's malloc/free
    // which respects ALLOW_MEMORY_GROWTH. page_allocator uses wasm memory.grow
    // directly and conflicts with emscripten's memory management.
    const allocator = std.heap.c_allocator;
    g_allocator = allocator;

{{hidden_setup}}    window.initWindow(screen_w, screen_h, screen_title);
    // Note: no defer closeWindow — Emscripten keeps running after main returns.

    g = AssembledGame.init(allocator);

    g.setHooks(&hooks);

    g.setScreenHeight(@as(f32, @floatFromInt(screen_h)));
    g.getCamera().setPosition(0, @as(f32, @floatFromInt(screen_h)));

    // Pass fps=0 so emscripten drives the loop via `requestAnimationFrame`
    // (vsync-aligned, browser-friendly) instead of `setTimeout` — a non-zero
    // fps triggers the browser's "use 0 for the frame rate" warning
    // (labelle-bgfx#14). Frame timing here comes from `window.frameDuration()`,
    // so this does not affect the game's dt.
{{preview_setup}}{{setup_code}}    emscripten_set_main_loop(&gameFrame, 0, 1);
}
